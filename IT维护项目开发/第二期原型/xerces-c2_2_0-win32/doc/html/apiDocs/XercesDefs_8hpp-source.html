00373 // ---------------------------------------------------------------------------
00374 //  Some general typedefs that are defined for internal flexibility.
00375 //
00376 //  Note  that UTF16Ch is fixed at 16 bits, whereas XMLCh floats in size per
00377 //  platform, to whatever is the native wide char format there. UCS4Ch is
00378 //  fixed at 32 bits. The types we defined them in terms of are defined per
00379 //  compiler, using whatever types are the right ones for them to get these
00380 //  16/32 bit sizes.
00381 //
00382 // ---------------------------------------------------------------------------
00383 typedef unsigned char       XMLByte;
00384 typedef XMLUInt16           UTF16Ch;
00385 typedef XMLUInt32           UCS4Ch;

00388 // ---------------------------------------------------------------------------
00389 //  Handle boolean. If the platform can handle booleans itself, then we
00390 //  map our boolean type to the native type. Otherwise we create a default
00391 //  one as an int and define const values for true and false.
00392 //
00393 //  This flag will be set in the per-development environment stuff above.
00394 // ---------------------------------------------------------------------------
00395 #if defined(NO_NATIVE_BOOL)
00396   #ifndef bool
00397     typedef int     bool;
00398   #endif
00399   #ifndef true
00400     #define  true     1
00401   #endif
00402   #ifndef false
00403     #define false 0
00404   #endif
00405 #endif
